# for finding out center of mass
def centerOfMass(img):
    im2,contours,hierarchy = cv2.findContours(img, 1, 2)
    cx = 0
    cy = 0
    area = 0
    for c in contours:
        m = cv2.moments(c)
        a = cv2.contourArea(c)
        cx = int((area * cx + (a*m['m10']/(m['m00']+0.00000001)))//(a + area+0.00000001))
        cy = int((area * cy + (a*m['m01']/(m['m00']+0.00000001)))//(a+area+0.00000001))
        area += a
    point = (cx,cy)
    
    return point , area

    point , Area = centerOfMass(imgs[i])

#butterworth filter
def dist( x1 , y1 , x2 , y2):
    return (math.sqrt((x1-x2)**2+(y1-y2)**2))
def butterworth_lpf(img ,d, order):
    rows , cols = img.shape
    cx , cy = rows//2 , cols//2
    filt = np.zeros((rows,cols) , dtype=np.float32)
    for x in range(rows):
        for y in range(cols):
            filt[x][y]=dist(x,y,cx,cy)
    #
    H =1/(1+(filt/d)**order)
    filtered = H * img
    
    return filtered
